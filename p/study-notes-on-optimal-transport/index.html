def compute_optimal_transport(M, r, c, lam, epsilon=1e-8):
    """
    Computes the optimal transport matrix and Sinkhorn distance using 
    the Sinkhorn-Knopp algorithm
    Inputs:
        - M : cost matrix (n, m)
        - r : vector of marginals (n, )
        - c : vector of marginals (m, )
        - lam : strength of the entropic regularization
        - epsilon : convergence parameter
    Outputs:
        - P : optimal transport matrix (n, m)
        - dist : Sinkhorn distance
    """
    n, m = M.shape
    # initialize: P = e^{-\lambda * M} (P_{ij} = e^{-\lambda * M_{ij}})
    P = np.exp(- lam * M)
    P /= P.sum()
    u = np.zeros(n)
    # normalize this matrix
    while np.max(np.abs(u - P.sum(1))) > epsilon:
        # scale the rows such that the row sums match r
        u = P.sum(1)
        P *= (r / u).reshape((-1, 1)) 
        # scale the columns such that the column sums match c
        P *= (c / P.sum(0)).reshape((1, -1))
    return P, np.sum(P * M)
